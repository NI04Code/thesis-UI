Algorithm 1: Compute Placement Cost algorithm \\
\vspace{-0.5em}
\rule{\linewidth}{0.4pt} \\
\textbf{Input:} \textit{candidateNode}, \textit{D}, \textit{nodesMap}, \textit{costProvider} \\
\textbf{Output:} \textit{totalCost}
\begin{algorithmic}[1]
\STATE $totalCost \leftarrow 0$
\FOR{\textbf{each} pod $d \in D$}
    \STATE $depNode \leftarrow nodesMap[d.nodeName]$
    \IF{$depNode$ \textbf{not found}}
        \STATE \textbf{continue}
    \ENDIF
    \STATE $totalCost \leftarrow totalCost + Cost(candidateNode, depNode)$
\ENDFOR
\RETURN $totalCost$
\end{algorithmic}
